# D

- 打一个前缀异或和，然后同 C，注意在开始时加入一个0先，代表前 $0$ 个数异或和是 $0$
- 有以下式子
$$(a_{p} \oplus a_{p+1} \oplus...\oplus a_{N})\oplus x=(a_{1}\oplus a_{2}\oplus...\oplus a_{N})\oplus x\oplus(a_1\oplus...\oplus a_{p-1})$$

```cpp
#define f(i,a,b) for(int i=a;i<=b;i++)
#define mid (l+r)/2
using namespace std;
typedef long long ll;
const int N=2e5+5;
int t=1;

int n,m;
int d[N<<6];
int a[N];
int rt[N<<5];
int node=0;
int son[N<<6][2];

void modify(int pre,int &x,int val,int dep){
    x=++node;
    d[x]=d[pre]+1;
    son[x][0]=son[pre][0];
    son[x][1]=son[pre][1];
    if(!dep) return;
    dep--;
    int nx=(val>>dep)&1;
    modify(son[pre][nx],son[x][nx],val,dep);
}

int query(int x,int y,int val,int dep){
    if(!dep) return 0;
    dep--;
    int nx=(val>>dep)&1;
    if(d[son[y][nx^1]]-d[son[x][nx^1]]) return (1<<dep)+query(son[x][nx^1],son[y][nx^1],val,dep);
    else return query(son[x][nx],son[y][nx],val,dep);
}

void solve(){
    cin>>n>>m;
    modify(rt[0],rt[0],0,24);
    int res=0;
    f(i,1,n){
        cin>>a[i];
        modify(rt[i-1],rt[i],res^=a[i],24);

    while(m--){
        char A;

        cin>>A;

            cin>>x;

        }
        else {
            cin>>l>>r>>x;
            if(l==1) l=0;
            else l=rt[l-2];
            r=rt[r-1];
            cout<<query(l,r,x^res,24)<<"\n";
        }
    }
}



int main(){
    std::ios::sync_with_stdio(0);
    std::cin.tie(0);
    // cin>>t;

    return 0;
```

```cpp
	if(dep[x]<dep[y]) swap(x,y);

		if(x==y) return x;
```

# I

- 我们设 $f_{i,j}$ 为数字 $j$ 在位置 $i$ 之前出现最后一次的位置，求 $\mathbf{mex}$ 既是求对于找到最小的 $v$ 使得 $f_{r,v}<l$ ，依旧可持久化线段树维护

```cpp
#define f(i,a,b) for(int i=a;i<=b;i++)
#define df(i,a,b) for(int i=a;i>=b;i--)
#define mid (l+r)/2
using namespace std;
typedef long long ll;
const int N=5e5+10;
int TT=1;

int d[N<<6],ls[N<<6],rs[N<<6];
int a[N],rt[N];
int node=0;
int n,m;

void modify(int pre,int &x,int l,int r,int pos,int v){
    x=++node;
    ls[x]=ls[pre],rs[x]=rs[pre];
    if(l==r){
        d[x]=v;
        return ;
    }

    if(pos<=mid) modify(ls[pre],ls[x],l,mid,pos,v);
    else modify(rs[pre],rs[x],mid+1,r,pos,v);

    d[x]=min(d[ls[x]],d[rs[x]]);
}

int query(int x,int l,int r,int v){
    if(l==r) return l;
    if(d[ls[x]]>=v) return query(rs[x],mid+1,r,v);
    else return query(ls[x],l,mid,v);
}

void solve(){
    cin>>n>>m;
    f(i,1,n) cin>>a[i],modify(rt[i-1],rt[i],1,2e5+5,++a[i],i);

    while(m--){
        int l,r;
        cin>>l>>r;
        cout<<query(rt[r],1,2e5+5,l)-1<<"\n";
    }
}



int main(){
    std::ios::sync_with_stdio(0);
    std::cin.tie(0);
    // cin>>TT;

    return 0;
}
```

# J


- 这道题不一样的数据有不一样的写法，对于 $r\ne 1$ 时的写法就不讲了，当 $n=1$ 时用可持久化线段树维护权值线段树上的区间和计数和，然后用类似求区间第 $k$ 大的方法来二分求解

```cpp
#define f(i,a,b) for(int i=a;i<=b;i++)
#define df(i,a,b) for(ll i=a;i>=b;i--)
#define mid (l+r)/2
using namespace std;
typedef long long ll;
const int N=5e5+10;
int TT=1;

int n,m,q;

int rt[N],ls[N<<6],rs[N<<6];
ll d[N<<6],cnt[N<<6];
int node=0;

void modify(int pre,int &x,int l,int r,int pos){
    x=++node;
    d[x]=d[pre]+pos;
    cnt[x]=cnt[pre]+1;
    ls[x]=ls[pre],rs[x]=rs[pre];
    if(l==r) return ;
    if(pos<=mid) modify(ls[pre],ls[x],l,mid,pos);
    else modify(rs[pre],rs[x],mid+1,r,pos);
}

int query(int x,int y,ll l,ll r,ll val){
    if(val<=0) return 0;
    if(l==r){
        return min(cnt[y]-cnt[x],(val+l-1)/l);
    }

    if(d[rs[y]]-d[rs[x]]<=val) return cnt[rs[y]]-cnt[rs[x]]+query(ls[x],ls[y],l,mid,val-(d[rs[y]]-d[rs[x]]));
    else return query(rs[x],rs[y],mid+1,r,val);
}

ll D[205][205][1000];
ll P[205][205];
ll maxn=0;

void work1(){
    f(i,1,m){
        int x;
        cin>>x;
        modify(rt[i-1],rt[i],1,1000,x);
    }
    while(q--){
        int l,r,x,y;
        ll H;
        cin>>x>>l>>y>>r>>H;
        if(d[rt[r]]-d[rt[l-1]]<H) cout<<"Poor QLW\n";
        else cout<<query(rt[l-1],rt[r],1,1000,H)<<"\n";
    }
}

void work2(){
    f(i,1,n){
        f(j,1,m){
            cin>>P[i][j];
            maxn=max(maxn,P[i][j]);
        }
    }

    f(i,1,n){
        f(j,1,m){
            f(k,1,maxn) D[i][j][k]=D[i][j-1][k]+D[i-1][j][k]-D[i-1][j-1][k];
            D[i][j][P[i][j]]++;
        }
    }

    while(q--){
        int x1,y1,x2,y2;
        ll H;
        cin>>x1>>y1>>x2>>y2>>H;
        int cnt=0;
        df(k,maxn,1){
            ll res=min(D[x2][y2][k]+D[x1-1][y1-1][k]-D[x2][y1-1][k]-D[x1-1][y2][k],(H+k-1)/k);
            cnt+=res;
            H-=res*k;
            if(H<=0) break;
        }
        if(H>0) cout<<"Poor QLW\n";
        else cout<<cnt<<"\n";
    }
}


void solve(){
    cin>>n>>m>>q;
    if(n==1) work1();
    else work2();
}



int main(){
    while(TT--) solve();
    return 0;
}
```

# K

- 用可持久化线段树继承的性质来差分，然后就是类似求区间第 $k$ 小来求和

```cpp
#define f(i,a,b) for(int i=a;i<=b;i++)
#define df(i,a,b) for(int i=a;i>=b;i--)
#define mid (l+r)/2
using namespace std;
typedef long long ll;
const int N=2e5+10;
const ll Inf=1e7+5;
int TT=1;

int rt[N];
ll cnt[N<<6],val[N<<6];
int ls[N<<6],rs[N<<6];
int node;
int n,m;

vector <ll> r[N];

void modify(int pre,int &x,int l,int r,int pos,ll v){
    x=++node;
    cnt[x]=cnt[pre];
    val[x]=val[pre];
    ls[x]=ls[pre];
    rs[x]=rs[pre];

    if(l==r){
        if(v==1) cnt[x]++,val[x]+=pos;
        else cnt[x]--,val[x]-=pos;
        return ;
    }

    if(pos<=mid) modify(ls[pre],ls[x],l,mid,pos,v);
    else modify(rs[pre],rs[x],mid+1,r,pos,v);

    val[x]=val[ls[x]]+val[rs[x]];
    cnt[x]=cnt[ls[x]]+cnt[rs[x]];
}

ll query(int x,int l,int r,ll k){
    if(l==r){
        return min(k,cnt[x])*l;
    }

    if(k<=cnt[ls[x]]) return query(ls[x],l,mid,k);
    else return val[ls[x]]+query(rs[x],mid+1,r,k-cnt[ls[x]]);
}


void solve(){
    cin>>m>>n;
    f(i,1,m){
        int s,e;
        ll p;
        cin>>s>>e>>p;
        r[s].push_back(p);
        r[e+1].push_back(-p);
    }

    f(i,1,n){
        rt[i]=rt[i-1];
        for(int it:r[i]){
            modify(rt[i],rt[i],1,Inf,abs(it),it/abs(it));
        }
    }

    ll pre=1;

    for(ll i=1;i<=n;i++){
        ll x,a,b,c;
        cin>>x>>a>>b>>c;
        ll k=1+(a*pre+b)%c;
        ll res=query(rt[x],1,Inf,k);
        pre=res;
        cout<<res<<"\n";
    }
}



int main(){
    std::ios::sync_with_stdio(0);
    std::cin.tie(0);
    // cin>>TT;
    while(TT--) solve();
    return 0;
}
```

# L

- 注意到如果一个数出现超过 $k$ 次，我们从第一个开始 $k$ 个 $k$ 个地取一定会取到这个数，然后根据 $\frac{r-l+1}{k}$ 较大，我们可以取第 $rk$ 小，第 $rk*2$ 大等等来看其出现次数有没有超过要求次数

```cpp

    else modify(rs[pre],rs[x],mid+1,r,pos);
```

# M

- 我们从高位往低位贪心，假如我们现在已经得到的答案时 $\mathbf{cur}$ ，如果当前为 $0$ ，我们目标就是 $i$ ，既在区间中找存不存在 $[cur+2^{i}-x,cur+2^{i+1}-x)$ 的数，反之找 $[cur-x,cur+2^{i}-x)$

# O

- 先按照成绩排序，然后从成绩大到小枚举，看能不能在左右区间各取 $\frac{n-1}{2}$ 个数出来，如果可以的话各取前 $\frac{n-1}{2}$ 小，然后看是不是小于 $f$ 即可，总复杂度 $O(n\log n)$ 

```cpp
#define f(i,a,b) for(int i=a;i<=b;i++)
#define df(i,a,b) for(int i=a;i>=b;i--)
#define mid (l+r)/2
using namespace std;
typedef long long ll;
const int N=2e5+10;
int TT=1;

int c,n;

ll d[N<<6],cnt[N<<6];
int rt[N],ls[N<<6],rs[N<<6];
int node=0;

void modify(int pre,int &x,ll l,ll r,ll pos){
    x=++node;
    d[x]=d[pre]+pos;
    cnt[x]=cnt[pre]+1;
    ls[x]=ls[pre],rs[x]=rs[pre];
    if(l==r) return ;

    if(pos<=mid) modify(ls[pre],ls[x],l,mid,pos);
    else modify(rs[pre],rs[x],mid+1,r,pos);
}

ll query(int x,int y,ll l,ll r,ll rk){
    if(rk<=0) return 0;
    if(l==r){
        return rk*l;
    }

    int vk=cnt[ls[y]]-cnt[ls[x]];

    if(rk<=vk) return query(ls[x],ls[y],l,mid,rk);
    else return d[ls[y]]-d[ls[x]]+query(rs[x],rs[y],mid+1,r,rk-vk);
}

ll F;
pair <ll,ll> a[N];

void solve(){
    cin>>n>>c>>F;
    f(i,1,c){
        ll x,y;
        cin>>x>>y;
        a[i]={x,y};
    }
```
